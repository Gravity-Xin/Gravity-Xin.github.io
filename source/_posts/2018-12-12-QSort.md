## 实现

```Go
func QSort(slice []int) {
    if len(slice) <= 1 {
        return
    }
    i := 0
    j := len(slice) - 1
    key := slice[0]
    for j > i {
        for slice[j] >= key && j > i {
            j--
        }
        for slice[i] < key && j > i {
            i++
        }
        slice[i], slice[j] = slice[j], slice[i]
    }
    j++
    qSort(slice[:j])
    qSort(slice[j:])
}
```
